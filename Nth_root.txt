Input: N = 3, M = 27
Output: 3
Explanation: The cube root of 27 is equal to 3.
Input : N = 4, M = 69
Output: -1
Explanation : The 4th root of 69 does not exist. So, the answer is -1.



import java.util.*;

class Main {

    public static int NthRoot(int n, int m) {

        int low = 1;
        int high = m;

        while (low <= high) {

            int mid = low + (high - low) / 2;

            long value = power(mid, n);

            if (value == m)
                return mid;

            else if (value < m)
                low = mid + 1;

            else
                high = mid - 1;
        }

        return -1;
    }

    public static long power(int base, int e) {

        long ans = 1;

        for (int i = 0; i < e; i++) {

            ans = ans * base;
        }

        return ans;
    }

    public static void main(String args[]) {

        int n = 4;
        int m = 69;

        int result = NthRoot(n, m);

        System.out.print(result);
    }
}
